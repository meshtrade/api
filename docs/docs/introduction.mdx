Welcome to the **Mesh API Documentation** - your comprehensive guide to integrating with the Mesh trading platform.

## What is Mesh API?

Mesh API is a modern, gRPC-based financial trading platform API that provides access to:

- **Identity & Access Management** - Role-based access control and API user management
- **Trading Services** - Direct orders, limit orders, and spot trading *(coming soon)*
- **Compliance Services** - Client onboarding, KYC, and regulatory compliance *(coming soon)*
- **Wallet Management** - Account creation and balance management *(coming soon)*
- **Issuance Hub** - Financial instrument creation and management *(coming soon)*

## Quick Start

Let's get you up and running with **Mesh API in 5 minutes** using the API User Service.

### 1. Installation

Install the SDK for your preferred language:

### 2. Get Your API Key

To use the Mesh API, you'll need an API key and group ID:

1. **Contact** the Mesh team to set up your account
2. **Receive** your initial API key and group ID
3. **Store** them securely as environment variables

:::tip
Keep your API key secure and never commit it to version control. Use environment variables or secure configuration management.
:::

### 3. Your First API Request

Let's create and manage an API user to test your setup:

<Tabs>
<TabItem value="go" label="Go">

```go
package main

import (
    "context"
    "log"

    api_userv1 "github.com/meshtrade/api/go/iam/api_user/v1"
    optionv1 "github.com/meshtrade/api/go/option/v1"
)

func main() {
    // Create API User Service client
    client, err := api_userv1.NewApiUserServiceGRPCClient(
        api_userv1.WithAPIKey("your-api-key"),
        api_userv1.WithGroup("your-group-id"),
        api_userv1.WithAddress("localhost", 8080), // Replace with actual endpoint
        api_userv1.WithTLS(false), // Use true in production
    )
    if err != nil {
        log.Fatalf("Failed to create client: %v", err)
    }
    defer client.Close()

    // Create a new API user
    apiUser, err := client.CreateApiUser(
        context.Background(),
        &api_userv1.CreateApiUserRequest{
            ApiUser: &api_userv1.APIUser{
                Owner:       "groups/your-group-id",
                DisplayName: "My First API User",
                Roles: []optionv1.Role{
                    optionv1.Role_ROLE_IAM_VIEWER,
                },
            },
        },
    )
    if err != nil {
        log.Fatalf("Failed to create API user: %v", err)
    }

    log.Printf("Created API user: %s", apiUser.Name)
    log.Printf("API key: %s", apiUser.ApiKey) // Save this securely!
}
```

</TabItem>
<TabItem value="python" label="Python">

```python
import grpc
from meshtrade.iam.api_user.v1 import service_pb2, service_pb2_grpc, api_user_pb2
from meshtrade.option.v1 import role_pb2

def main():
    # Create gRPC channel and client
    channel = grpc.insecure_channel('localhost:8080')  # Replace with actual endpoint
    client = service_pb2_grpc.ApiUserServiceStub(channel)
    
    # Authentication metadata
    metadata = [
        ('authorization', 'Bearer your-api-key'),
        ('x-group-id', 'your-group-id'),
    ]
    
    try:
        # Create a new API user
        api_user = api_user_pb2.APIUser(
            owner="groups/your-group-id",
            display_name="My First API User",
            roles=[role_pb2.Role.ROLE_IAM_VIEWER]
        )
        
        request = service_pb2.CreateApiUserRequest(api_user=api_user)
        response = client.CreateApiUser(request, metadata=metadata)
        
        print(f"Created API user: {response.name}")
        print(f"API key: {response.api_key}")  # Save this securely!
        
    except grpc.RpcError as e:
        print(f"Failed to create API user: {e}")
    
    finally:
        channel.close()

if __name__ == "__main__":
    main()
```

</TabItem>
</Tabs>

